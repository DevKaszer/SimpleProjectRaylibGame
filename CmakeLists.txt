cmake_minimum_required(VERSION 3.15)
project(PierwszaGra LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Znajdź wszystkie pliki .cpp w src/
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS
    src/*.cpp
)

# Tworzenie pliku wykonywalnego
add_executable(PierwszaGra ${SOURCES})

# Dodanie folderu include/ jako globalnego dla całego projektu
target_include_directories(PierwszaGra PRIVATE
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/libs/raylib/include
)

# Linkowanie do Raylib
target_link_directories(PierwszaGra PRIVATE
    ${CMAKE_SOURCE_DIR}/libs/raylib/lib
)

target_link_libraries(PierwszaGra PRIVATE
    raylib
    winmm
    gdi32
    opengl32
)
